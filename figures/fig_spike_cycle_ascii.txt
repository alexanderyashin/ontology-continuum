  \centering
  % ASCII diagram in figures/fig_spike_cycle_ascii.txt
  \caption{Neuronal spike cycle as a minimal time-carrying cycle in $K_5$. 
  The horizontal axis represents time, the vertical axis represents membrane
  potential $V(t)$. 
  The curve illustrates the sequence rest $\rightarrow$ depolarisation
  $\rightarrow$ overshoot $\rightarrow$ repolarisation $\rightarrow$
  hyperpolarisation and return to rest. 
  Crossing the excitation threshold $\Theta_{\mathrm{spike}}=V_{\mathrm{thr}}$
  corresponds to a local phase transition, and the full loop defines a cycle
  $C_{\mathrm{spike}}$ with period $\tau_{\mathrm{spike}}$. 
  Stable repetition of such cycles yields a non-zero $k_5$ and provides the
  physical substrate for higher-level cognitive dynamics in $K_6$.}
  \label{fig:spike-cycle}
\end{figure}
